// Example: Simple M5Tab5 App
#include "base_app.h"

class MyApp : public BaseApp {
public:
    MyApp() : BaseApp("com.example.myapp", 
                     "My Application", "1.0.0") {}
    
    os_error_t initialize() override {
        setDescription("Custom app for M5Stack Tab5");
        setMemoryUsage(32768); // 32KB
        return OS_OK;
    }
    
    os_error_t createUI(lv_obj_t* parent) override {
        m_container = lv_obj_create(parent);
        lv_obj_set_size(m_container, LV_PCT(100), LV_PCT(100));
        
        lv_obj_t* label = lv_label_create(m_container);
        lv_label_set_text(label, "Hello M5Stack Tab5!");
        lv_obj_center(label);
        
        return OS_OK;
    }
};

extern "C" std::unique_ptr<BaseApp> createMyApp() {
    return std::make_unique<MyApp>();
}
